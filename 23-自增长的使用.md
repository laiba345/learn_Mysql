自增长的使用
```
-- 演示自增长的使用
-- 创建表
CREATE TABLE t13(
	id INT PRIMARY KEY AUTO_INCREMENT,
	email VARCHAR(32) NOT NULL DEFAULT '', 
	`name` VARCHAR(32) NOT NULL DEFAULT ''); 
DESC t13; 
-- 测试自增长的使用
-- 1. 第一种形式
INSERT INTO t13 VALUES(NULL, 'kk@qq.com', 'tom'); 
-- 2. 第二种形式
INSERT INTO t13 (email, `name`) VALUES('hk@qq.com', 'hk'); 
SELECT * FROM t13; 
```
